def bubble_sort(a, N):
    for i in range(N-1, 0, -1): # 매단계 마지막 인덱스에 정렬된 값을 저장하기 때문에 구간을 1씩 줄임
        for j in range(i):
            if a[j] > a[j+1]: # 오름차순 정렬
                a[j+1], a[j] = a[j], a[j+1]
    return a
