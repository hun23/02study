def bubble_sort(a, N):
    i = 0
    while i < N:
        if a[i -1] > a[i]:   # 앞자리와 비교했을 때, 앞자리가 더 크다면 위치 바꾸기
            a[i - 1], a[i] = a[i], a[i-1]
            i += 1
        else:
            i -= 1  # 바꾸지 않았다면 탐색 위치 돌려 놓기

    return a

a = [55, 7, 78, 12, 42]
N = 5
print(bubble_sort(a, N))
